## Deployment


=== ":material-docker: Podman Compose"

    ```yaml
    services:
      grimmory:
        image: @REGISTRY@/grimmory:latest
        container_name: grimmory
        environment:
          - PUID=@PUID@
          - PGID=@PGID@
          - TZ=@TZ@
          - SPRING_DATASOURCE_URL=jdbc:mariadb://127.0.0.1:3306/grimmory
          - SPRING_DATASOURCE_USERNAME=grimmory
          - SPRING_DATASOURCE_PASSWORD=changeme
        volumes:
          - "@CONTAINER_CONFIG_ROOT@/@GRIMMORY_APP_DATA_PATH@:/app/data"
          - "/path/to/@BOOKS_PATH@:/books"
          - "@CONTAINER_CONFIG_ROOT@/@GRIMMORY_BOOKDROP_PATH@:/bookdrop"
        restart: unless-stopped
    ```
